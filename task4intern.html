from pulp import LpMaximize, LpProblem, LpVariable, lpSum
In[3]:
# Define the linear programming problem
problem = LpProblem("Maximize_Profit", LpMaximize)
In[4]:
# Decision variables: Number of units to produce for each product
x1 = LpVariable("Product_A", lowBound=0, cat="Continuous")
x2 = LpVariable("Product_B", lowBound=0, cat="Continuous")
In[5]:
# Objective function: Maximize profit
problem += 40 * x1 + 30 * x2, "Total_Profit"
In[6]:
# Constraints:
# Machine 1 usage
problem += 2 * x1 + 1 * x2 <= 120, "Machine_1_Hours"

# Machine 2 usage
problem += 1 * x1 + 1 * x2 <= 80, "Machine_2_Hours"
In[7]:
# Solve the problem
problem.solve()
Out[7]:
1

print(f"Status: {problem.status}")
print(f"Optimal number of Product A to produce: {x1.varValue}")

print(f"Maximum profit: ${problem.objective.value()}")
